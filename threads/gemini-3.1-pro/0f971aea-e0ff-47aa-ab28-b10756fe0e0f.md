  T1.ID,
  CAST(strftime('%Y', CURRENT_TIMESTAMP) AS INTEGER) - CAST(strftime('%Y', T1.Birthday) AS INTEGER) AS age

JOIN Laboratory AS T2

GROUP BY
  T1.ID
HAVING
