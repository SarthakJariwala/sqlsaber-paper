SELECT 
    d.forename || ' ' || d.surname AS full_name,
    SUM(CAST(p.duration AS REAL)) / COUNT(p.duration) AS avg_duration
FROM drivers d
JOIN pitStops p ON d.driverId = p.driverId
WHERE d.nationality = 'German'
  AND CAST(strftime('%Y', d.dob) AS INTEGER) BETWEEN 1980 AND 1985
GROUP BY d.driverId, full_name
ORDER BY avg_duration ASC
LIMIT 3;
