WITH MostInfluential AS (
    SELECT Id 
    FROM users 
    ORDER BY Reputation DESC 
    LIMIT 1
)
SELECT 
    CAST(SUM(CASE WHEN p.Score > 50 THEN 1 ELSE 0 END) AS REAL) / COUNT(p.Id) AS pct
FROM posts p
JOIN MostInfluential u ON p.OwnerUserId = u.Id;
