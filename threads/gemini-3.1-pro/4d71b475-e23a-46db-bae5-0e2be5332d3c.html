SELECT
  T.team_long_name
FROM League AS L
JOIN Match AS M ON L.id = M.league_id
JOIN Team AS T ON M.away_team_api_id = T.team_api_id
WHERE L.name = 'Scotland Premier League'
  AND M.season = '2009/2010'
  AND M.away_team_goal > M.home_team_goal
GROUP BY T.team_api_id, T.team_long_name
ORDER BY COUNT(*) DESC
LIMIT 1;
