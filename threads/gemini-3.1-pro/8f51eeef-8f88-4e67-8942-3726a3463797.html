    c.name, 
    r.text AS ruling_text, 
    CASE 
        WHEN c.hasContentWarning = 1 THEN 'Yes' 
        ELSE 'No' 
    END AS has_missing_or_degraded_properties
FROM cards c
JOIN rulings r ON c.uuid = r.uuid
WHERE c.artist = 'Jim Pavelec'
ORDER BY c.name
