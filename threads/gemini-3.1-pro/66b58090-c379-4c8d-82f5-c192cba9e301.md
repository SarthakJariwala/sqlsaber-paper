SELECT 
  c.name AS circuit_name, 
  MIN(l.milliseconds) AS min_ms_laptimes,
  MIN(CAST(SUBSTR(r.fastestLapTime,1,INSTR(r.fastestLapTime,':')-1)*60000 + SUBSTR(r.fastestLapTime,INSTR(r.fastestLapTime,':')+1,2)*1000 + SUBSTR(r.fastestLapTime,INSTR(r.fastestLapTime,'.')+1,3) AS INTEGER)) AS min_ms_results
FROM circuits c 
JOIN races ra ON c.circuitId = ra.circuitId 
LEFT JOIN lapTimes l ON ra.raceId = l.raceId 
LEFT JOIN results r ON ra.raceId = r.raceId

GROUP BY c.name
