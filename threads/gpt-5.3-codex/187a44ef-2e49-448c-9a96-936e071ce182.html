WITH filtered AS (
  SELECT
    CDSCode,
    School,
    City,
    County,
    District,
    CAST(OpenDate AS TEXT) AS OpenDate_text,
    TRIM(AdmEmail1) AS AdmEmail1,
    TRIM(AdmEmail2) AS AdmEmail2,
    TRIM(AdmEmail3) AS AdmEmail3
  FROM schools
  WHERE County = 'San Bernardino'
    AND District = 'San Bernardino City Unified'
    AND DOC = '54'
    AND SOC = '62'
    AND date(OpenDate) BETWEEN date('2009-01-01') AND date('2010-12-31')
)
SELECT DISTINCT LOWER(email) AS administrator_email
FROM (
  SELECT AdmEmail1 AS email FROM filtered
  UNION ALL
  SELECT AdmEmail2 AS email FROM filtered
  UNION ALL
  SELECT AdmEmail3 AS email FROM filtered
)
WHERE email IS NOT NULL
  AND TRIM(email) <> ''
  AND email LIKE '%@%.%'
  AND email NOT LIKE '% %'
ORDER BY administrator_email
LIMIT 1000;
