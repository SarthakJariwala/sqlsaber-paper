WITH driver_best AS (
  SELECT
    lt.driverId,
    MIN(lt.milliseconds) AS best_ms
  FROM lapTimes lt
  WHERE lt.time IS NOT NULL
    AND lt.milliseconds IS NOT NULL
  GROUP BY lt.driverId
),
best_time_text AS (
  SELECT
    db.driverId,
    db.best_ms,
    MIN(lt.time) AS shortest_lap_time
  FROM driver_best db
  JOIN lapTimes lt
    ON lt.driverId = db.driverId
   AND lt.milliseconds = db.best_ms
  GROUP BY db.driverId, db.best_ms
)
SELECT
  d.forename || ' ' || d.surname AS full_name,
  b.shortest_lap_time,
  b.best_ms AS shortest_lap_milliseconds
FROM best_time_text b
JOIN drivers d
  ON d.driverId = b.driverId
ORDER BY b.best_ms ASC, full_name ASC
LIMIT 20;
