WITH yearly AS (
  SELECT
    CAST(strftime('%Y', CAST(l.date AS TEXT)) AS INTEGER) AS loan_year,
    SUM(l.amount) AS total_amount
  FROM main.loan l
  JOIN main.disp d
    ON d.account_id = l.account_id
   AND d.type = 'OWNER'
  JOIN main.client c
    ON c.client_id = d.client_id
  WHERE c.gender = 'M'
    AND CAST(strftime('%Y', CAST(l.date AS TEXT)) AS INTEGER) IN (1996, 1997)
  GROUP BY CAST(strftime('%Y', CAST(l.date AS TEXT)) AS INTEGER)
),
paired AS (
  SELECT
    MAX(CASE WHEN loan_year = 1996 THEN total_amount END) AS amount_1996,
    MAX(CASE WHEN loan_year = 1997 THEN total_amount END) AS amount_1997
  FROM yearly
)
SELECT
  amount_1996,
  amount_1997,
  ROUND(
    (amount_1997 - amount_1996) * 100.0 / NULLIF(amount_1996, 0),
    2
  ) AS growth_rate_percent
FROM paired
LIMIT 1;
