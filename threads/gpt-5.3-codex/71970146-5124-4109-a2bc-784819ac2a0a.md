            WHEN 'pb' THEN 'Lead'
            WHEN 'te' THEN 'Tellurium'
            WHEN 'ca' THEN 'Calcium'
            ELSE a.element
        END

FROM triple_molecules tm
JOIN molecule m ON m.molecule_id = tm.molecule_id

ORDER BY m.molecule_id
LIMIT 1000;
