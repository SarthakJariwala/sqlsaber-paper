```sql
WITH top_user AS (
  SELECT Id, DisplayName, Reputation
  FROM users
  ORDER BY Reputation DESC, Id ASC
  LIMIT 1
)
SELECT
  tu.Id AS user_id,
  tu.DisplayName AS display_name,
  tu.Reputation AS reputation,
  COUNT(p.Id) AS total_posts,
  SUM(CASE WHEN p.Score > 50 THEN 1 ELSE 0 END) AS posts_score_above_50,
  ROUND(
    100.0 * SUM(CASE WHEN p.Score > 50 THEN 1 ELSE 0 END) / NULLIF(COUNT(p.Id), 0),
    2
  ) AS percentage_above_50
FROM top_user tu
LEFT JOIN posts p
  ON p.OwnerUserId = tu.Id
GROUP BY tu.Id, tu.DisplayName, tu.Reputation
LIMIT 1000;
```
