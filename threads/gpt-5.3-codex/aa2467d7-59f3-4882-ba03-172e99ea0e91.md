)

    SUM(CASE WHEN LOWER(a.element) = 'cl' THEN 1 ELSE 0 END) * 1.0 / COUNT(a.atom_id) AS percent_ratio,

FROM atom a
JOIN single_bond_molecules sbm
  ON a.molecule_id = sbm.molecule_id
LIMIT 1;
