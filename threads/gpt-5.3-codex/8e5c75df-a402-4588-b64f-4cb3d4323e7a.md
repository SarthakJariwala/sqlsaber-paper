**Executing SQL:**

```sql
WITH winner_times AS (
  SELECT
    r.year,
    res.time,
    instr(res.time, ':') AS p1,
    substr(res.time, instr(res.time, ':') + 1) AS rem
  FROM results res
  JOIN races r ON r.raceId = res.raceId
  WHERE r.year < 1975
    AND res.positionOrder = 1
    AND res.time IS NOT NULL
), parsed AS (
  SELECT
    year,
    CAST(substr(time, 1, p1 - 1) AS REAL) AS hours,
    CAST(substr(rem, 1, instr(rem, ':') - 1) AS REAL) AS minutes,
    CAST(substr(rem, instr(rem, ':') + 1) AS REAL) AS seconds
  FROM winner_times
)
SELECT
  year,
  ROUND(AVG(hours * 3600 + minutes * 60 + seconds), 3) AS avg_champion_time_seconds,
  COUNT(*) AS champion_finishes_count
FROM parsed
GROUP BY year
ORDER BY year
LIMIT 1000;
```
