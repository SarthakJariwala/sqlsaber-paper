WITH student_badges AS (
  SELECT Name, CAST(Date AS TEXT) AS DateText
  FROM badges
  WHERE Name = 'Student'
),
counts AS (
  SELECT
    COUNT(Name) AS total_count,
    SUM(CASE WHEN strftime('%Y', DateText) = '2010' THEN 1 ELSE 0 END) AS count_2010,
    SUM(CASE WHEN strftime('%Y', DateText) = '2011' THEN 1 ELSE 0 END) AS count_2011
  FROM student_badges
)
SELECT
  count_2010,
  count_2011,
  total_count,
  ROUND((count_2010 * 100.0 / NULLIF(total_count, 0)), 6) AS pct_2010,
  ROUND((count_2011 * 100.0 / NULLIF(total_count, 0)), 6) AS pct_2011,
  ROUND(
    (count_2010 * 100.0 / NULLIF(total_count, 0)) -
    (count_2011 * 100.0 / NULLIF(total_count, 0)),
    6
  ) AS percentage_difference_2010_minus_2011
FROM counts
LIMIT 1;
