```sql
WITH segment_year AS (
  SELECT
    c.Segment,
    SUBSTR(CAST(y.Date AS TEXT), 1, 4) AS year,
    SUM(y.Consumption) AS total_consumption
  FROM yearmonth y
  JOIN customers c ON c.CustomerID = y.CustomerID
  WHERE c.Currency = 'EUR'
    AND c.Segment IN ('SME', 'LAM', 'KAM')
    AND SUBSTR(CAST(y.Date AS TEXT), 1, 4) IN ('2012', '2013')
  GROUP BY c.Segment, SUBSTR(CAST(y.Date AS TEXT), 1, 4)
),
pivoted AS (
  SELECT
    Segment,
    SUM(CASE WHEN year = '2012' THEN total_consumption ELSE 0 END) AS consumption_2012,
    SUM(CASE WHEN year = '2013' THEN total_consumption ELSE 0 END) AS consumption_2013
  FROM segment_year
  GROUP BY Segment
)
SELECT
  Segment,
  ROUND(consumption_2012, 2) AS consumption_2012,
  ROUND(consumption_2013, 2) AS consumption_2013,
  ROUND(consumption_2013 - consumption_2012, 2) AS increase_or_decrease,
  ROUND(((consumption_2013 - consumption_2012) / NULLIF(consumption_2013, 0)) * 100, 4) AS percentage_increase
FROM pivoted
ORDER BY percentage_increase DESC
LIMIT 10;
```
