```sql
WITH extremes AS (
  SELECT player_api_id, player_name, height,
         CASE
           WHEN height = (SELECT MAX(height) FROM Player) THEN 'Tallest'
           WHEN height = (SELECT MIN(height) FROM Player) THEN 'Shortest'
         END AS height_group
  FROM Player
  WHERE height IN ((SELECT MIN(height) FROM Player), (SELECT MAX(height) FROM Player))
), finishing_avg AS (
  SELECT
    e.height_group,
    e.player_name,
    e.height,
    AVG(pa.finishing) AS avg_finishing
  FROM extremes e
  JOIN Player_Attributes pa
    ON pa.player_api_id = e.player_api_id
  WHERE pa.finishing IS NOT NULL
  GROUP BY e.height_group, e.player_name, e.height
)
SELECT height_group, player_name, height, ROUND(avg_finishing, 2) AS avg_finishing
FROM finishing_avg
ORDER BY avg_finishing DESC
LIMIT 10;
```
