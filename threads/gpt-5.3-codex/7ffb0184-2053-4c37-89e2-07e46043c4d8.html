WITH ratings AS (
  SELECT p.player_name,
         pa.overall_rating
  FROM Player AS p
  JOIN Player_Attributes AS pa
    ON p.player_api_id = pa.player_api_id
  WHERE p.player_name IN ('Landon Donovan', 'Jordan Bowery')
    AND DATE(pa.date) = '2013-07-12'
)
SELECT MAX(CASE WHEN player_name = 'Landon Donovan' THEN overall_rating END) AS landon_overall_rating,
       MAX(CASE WHEN player_name = 'Jordan Bowery' THEN overall_rating END) AS jordan_overall_rating,
       ROUND(
         (
           (MAX(CASE WHEN player_name = 'Landon Donovan' THEN overall_rating END) -
            MAX(CASE WHEN player_name = 'Jordan Bowery' THEN overall_rating END)
           ) * 100.0
         ) / NULLIF(MAX(CASE WHEN player_name = 'Landon Donovan' THEN overall_rating END), 0),
         2
       ) AS percentage_higher
FROM ratings
LIMIT 1;
