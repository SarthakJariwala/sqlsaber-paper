```sql
WITH qualifying_members AS (
    SELECT
        e.link_to_member AS member_id,
        COUNT(DISTINCT b.link_to_event) AS event_count
    FROM expense e
    JOIN budget b
      ON e.link_to_budget = b.budget_id
    WHERE b.link_to_event IS NOT NULL
    GROUP BY e.link_to_member
    HAVING COUNT(DISTINCT b.link_to_event) > 1
),
member_max_cost AS (
    SELECT
        e.link_to_member AS member_id,
        MAX(e.cost) AS max_cost
    FROM expense e
    JOIN budget b
      ON e.link_to_budget = b.budget_id
    JOIN qualifying_members qm
      ON qm.member_id = e.link_to_member
    GROUP BY e.link_to_member
),
top_cost AS (
    SELECT MAX(max_cost) AS top_max_cost
    FROM member_max_cost
)
SELECT
    mmc.member_id,
    m.first_name,
    m.last_name,
    mmc.max_cost,
    qm.event_count
FROM member_max_cost mmc
JOIN top_cost tc
  ON mmc.max_cost = tc.top_max_cost
JOIN qualifying_members qm
  ON qm.member_id = mmc.member_id
LEFT JOIN member m
  ON m.member_id = mmc.member_id
ORDER BY mmc.member_id
LIMIT 100;
```
