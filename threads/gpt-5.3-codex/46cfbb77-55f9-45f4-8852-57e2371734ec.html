),
fastest_lap AS (
    SELECT 
        l.raceId,
        l.driverId,
        l.lap,
        l.time AS lap_time,
        l.milliseconds AS lap_ms,
        ROW_NUMBER() OVER (ORDER BY l.milliseconds ASC) AS rn

    JOIN austrian_gp_races ar ON ar.raceId = l.raceId

)

WITH austrian_gp_races AS (
    SELECT r.raceId
    FROM races r
    WHERE r.name = 'Austrian Grand Prix'
),
fastest_lap AS (
    SELECT l.raceId, l.driverId, l.milliseconds,
           ROW_NUMBER() OVER (ORDER BY l.milliseconds ASC) AS rn
    FROM lapTimes l
    JOIN austrian_gp_races ar ON ar.raceId = l.raceId
    WHERE l.milliseconds IS NOT NULL
)
SELECT 
    d.forename || ' ' || d.surname AS driver_name,
    COUNT(ps.stop) AS pit_stop_count,
    SUM(ps.milliseconds) AS total_pit_ms,
    ROUND(SUM(ps.milliseconds) / 1000.0, 3) AS total_pit_seconds
FROM fastest_lap fl
JOIN drivers d ON d.driverId = fl.driverId
LEFT JOIN pitStops ps ON ps.raceId = fl.raceId AND ps.driverId = fl.driverId
WHERE fl.rn = 1
GROUP BY d.driverId
LIMIT 10;
