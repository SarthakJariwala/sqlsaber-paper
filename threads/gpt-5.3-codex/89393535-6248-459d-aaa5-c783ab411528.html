WITH locally_funded AS (
    SELECT
        f.CDSCode,
        f."School Name" AS school_name,
        s.DOCType AS doc_type,
        (f."Enrollment (K-12)" - f."Enrollment (Ages 5-17)") AS enrollment_diff
    FROM frpm AS f
    LEFT JOIN schools AS s
        ON s.CDSCode = f.CDSCode
    WHERE f."Charter Funding Type" = 'Locally funded'
      AND f."Enrollment (K-12)" IS NOT NULL
      AND f."Enrollment (Ages 5-17)" IS NOT NULL
),
avg_diff AS (
    SELECT AVG(enrollment_diff) AS avg_enrollment_diff
    FROM locally_funded
)
SELECT
    lf.school_name AS "School Name",
    lf.doc_type AS "DOC Type"
FROM locally_funded AS lf
CROSS JOIN avg_diff AS a
WHERE lf.enrollment_diff > a.avg_enrollment_diff
ORDER BY lf.enrollment_diff DESC, lf.school_name
LIMIT 1000;
