  (CAST(strftime('%Y', 'now') AS INTEGER) - CAST(strftime('%Y', p.Birthday) AS INTEGER)) AS age_years,
  CASE

  COUNT(*) AS abnormal_rbc_records,
  MIN(l.RBC) AS min_rbc,

FROM Patient p

  AND l.RBC IS NOT NULL

GROUP BY p.ID, p.SEX, p.Birthday, p.Admission
ORDER BY p.ID
LIMIT 1000;
