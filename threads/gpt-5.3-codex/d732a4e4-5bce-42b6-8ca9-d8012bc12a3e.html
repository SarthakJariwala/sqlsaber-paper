SELECT DISTINCT
  e.event_name,
  CAST(e.event_date AS TEXT) AS event_date
FROM event e
JOIN budget b
  ON b.link_to_event = e.event_id
JOIN expense x
  ON x.link_to_budget = b.budget_id
WHERE x.expense_description = 'Pizza'
  AND x.cost > 50
  AND x.cost < 100
ORDER BY e.event_date
LIMIT 100;
