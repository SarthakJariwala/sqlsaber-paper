SELECT
  t.card_name,
  GROUP_CONCAT(t.format, ', ') AS legal_formats
FROM (
  SELECT DISTINCT
    c.name AS card_name,
    l.format AS format
  FROM cards c
  JOIN sets s ON s.code = c.setCode
  JOIN legalities l ON l.uuid = c.uuid
  WHERE s.name = 'Hour of Devastation'
    AND l.status = 'Legal'
  ORDER BY c.name, l.format
) AS t
GROUP BY t.card_name
ORDER BY t.card_name
LIMIT 1000;
