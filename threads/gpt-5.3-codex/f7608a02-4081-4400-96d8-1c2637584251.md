Now I’ll run one `SELECT` that computes the annual average set count (2012–2015) and the top language.


**Executing SQL:**

```sql
WITH filtered_sets AS (
    SELECT code,
           CAST(strftime('%Y', releaseDate) AS INTEGER) AS release_year
    FROM sets
    WHERE date(releaseDate) BETWEEN date('2012-01-01') AND date('2015-12-31')
),
yearly_set_counts AS (
    SELECT release_year,
           COUNT(*) AS sets_released
    FROM filtered_sets
    GROUP BY release_year
),
common_language AS (
    SELECT fd.language,
           COUNT(*) AS language_count
    FROM filtered_sets fs
    JOIN cards c
      ON c.setCode = fs.code
    JOIN foreign_data fd
      ON fd.uuid = c.uuid
    WHERE fd.language IS NOT NULL
      AND TRIM(fd.language) <> ''
    GROUP BY fd.language
    ORDER BY language_count DESC
    LIMIT 1
)
SELECT ROUND(AVG(y.sets_released), 2) AS annual_avg_sets_released,
       cl.language AS most_common_card_language,
       cl.language_count AS language_occurrences
FROM yearly_set_counts y
CROSS JOIN common_language cl
LIMIT 1;
```
