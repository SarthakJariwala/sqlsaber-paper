SELECT
  c.name AS card_name,
  c.setCode AS set_code,
  c.uuid AS card_uuid,
  CASE
    WHEN c.hasContentWarning = 1 THEN 'Yes'
    ELSE 'No'
  END AS has_missing_or_degraded_properties_values,
  COALESCE(
    GROUP_CONCAT(
      '[' || CAST(r.date AS TEXT) || '] ' || r.text,
      ' || '
    ),
    'No ruling text found'
  ) AS ruling_text
FROM cards c
LEFT JOIN rulings r
  ON r.uuid = c.uuid
WHERE c.artist = 'Jim Pavelec'
GROUP BY c.uuid, c.name, c.setCode, c.hasContentWarning
ORDER BY c.name, c.setCode
LIMIT 1000;
