WITH annual_consumption AS (

        c.Segment,
        y.CustomerID,
        SUM(y.Consumption) AS annual_consumption
    FROM yearmonth y

      ON c.CustomerID = y.CustomerID
    WHERE c.Currency = 'CZK'
      AND c.Segment IN ('SME', 'LAM', 'KAM')
      AND y.Date BETWEEN '201301' AND '201312'
    GROUP BY c.Segment, y.CustomerID
),
min_per_segment AS (
    SELECT
        Segment,
        MIN(annual_consumption) AS min_annual_consumption
    FROM annual_consumption
    GROUP BY Segment
),
lowest_customers AS (
    SELECT
        a.Segment,
        a.CustomerID,
        a.annual_consumption
    FROM annual_consumption a
    JOIN min_per_segment m
      ON a.Segment = m.Segment
     AND a.annual_consumption = m.min_annual_consumption
),
annual_avg_lowest AS (
    SELECT
        Segment,
        SUM(annual_consumption) * 1.0 / COUNT(*) AS annual_avg_consumption_lowest,
        COUNT(*) AS num_lowest_customers
    FROM lowest_customers
    GROUP BY Segment
),
pivoted AS (
    SELECT
        MAX(CASE WHEN Segment = 'SME' THEN annual_avg_consumption_lowest END) AS sme_avg,
        MAX(CASE WHEN Segment = 'LAM' THEN annual_avg_consumption_lowest END) AS lam_avg,
        MAX(CASE WHEN Segment = 'KAM' THEN annual_avg_consumption_lowest END) AS kam_avg,
        MAX(CASE WHEN Segment = 'SME' THEN num_lowest_customers END) AS sme_n,
        MAX(CASE WHEN Segment = 'LAM' THEN num_lowest_customers END) AS lam_n,
        MAX(CASE WHEN Segment = 'KAM' THEN num_lowest_customers END) AS kam_n
    FROM annual_avg_lowest
)
SELECT
    sme_avg,
    lam_avg,
    kam_avg,
    sme_n,
    lam_n,

FROM pivoted
LIMIT 1;
