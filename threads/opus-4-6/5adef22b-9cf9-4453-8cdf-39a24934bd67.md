**Executing SQL:**

```sql

SELECT 
  CAST(COUNT(CASE 
    WHEN CAST(STRFTIME('%Y', d.dob) AS INTEGER) < 1985 AND r.laps > 50 
    THEN r.driverId 
  END) AS REAL) * 100 
  / COUNT(r.driverId) AS percentage
FROM results r
INNER JOIN races ra ON r.raceId = ra.raceId
INNER JOIN drivers d ON r.driverId = d.driverId
WHERE ra.year BETWEEN 2000 AND 2005;

```
