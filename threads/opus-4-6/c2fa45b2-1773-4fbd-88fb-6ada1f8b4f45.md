```sql

SELECT DISTINCT
    p.ID,
    p.SEX,
    p.Diagnosis,
    l.Date AS Lab_Date,
    l.`T-BIL`
FROM Patient p
INNER JOIN Laboratory l ON p.ID = l.ID
WHERE p.Admission = '-'
  AND l.Date LIKE '1991-10%'
  AND l.`T-BIL` < 2.0
ORDER BY p.ID;

```
