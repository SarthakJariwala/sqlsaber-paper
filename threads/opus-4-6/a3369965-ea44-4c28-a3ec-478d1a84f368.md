**Executing SQL:**

```sql

SELECT 
    COUNT(r.raceId) AS total_races,
    SUM(CASE WHEN res.position > 1 THEN 1 ELSE 0 END) AS not_first_count,
    ROUND(
        CAST(SUM(CASE WHEN res.position > 1 THEN 1 ELSE 0 END) AS REAL) 
        / CAST(COUNT(r.raceId) AS REAL) * 100, 
        2
    ) AS percentage_not_first
FROM results res
JOIN drivers d ON res.driverId = d.driverId
JOIN races r ON res.raceId = r.raceId
WHERE d.surname = 'Hamilton'
  AND r.year >= 2010;

```
