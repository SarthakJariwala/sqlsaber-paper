SELECT DISTINCT a.element
FROM atom a
JOIN connected c ON a.atom_id = c.atom_id
JOIN bond b ON c.bond_id = b.bond_id
WHERE b.bond_type = '='
  AND a.molecule_id = 'TR024'
ORDER BY a.element;
