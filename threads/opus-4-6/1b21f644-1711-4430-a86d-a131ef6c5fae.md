```sql

SELECT 
    ROUND(
        CAST(SUM(CASE WHEN a.alignment = 'Bad' THEN 1 ELSE 0 END) AS REAL) * 100 
        / COUNT(s.id), 2
    ) AS percentage_bad_alignment,
    SUM(CASE WHEN a.alignment = 'Bad' AND p.publisher_name = 'Marvel Comics' THEN 1 ELSE 0 END) AS marvel_bad_count
FROM superhero s
LEFT JOIN alignment a ON s.alignment_id = a.id
LEFT JOIN publisher p ON s.publisher_id = p.id;

```
