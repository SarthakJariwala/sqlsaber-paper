SELECT 
    b.bond_type,
    COUNT(b.bond_type) AS bond_count,
    m.label AS carcinogenic_label
FROM bond b
JOIN molecule m ON b.molecule_id = m.molecule_id
WHERE b.molecule_id = 'TR010'
GROUP BY b.bond_type, m.label
ORDER BY bond_count DESC
LIMIT 1;
