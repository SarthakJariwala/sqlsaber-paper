**Executing SQL:**

```sql

SELECT 
    c.name AS circuit_name,
    c.location,
    lt.time AS lap_record_time,
    lt.milliseconds,
    d.forename || ' ' || d.surname AS driver,
    r.year AS race_year
FROM circuits c
JOIN races r ON c.circuitId = r.circuitId
JOIN lapTimes lt ON r.raceId = lt.raceId
JOIN drivers d ON lt.driverId = d.driverId
WHERE c.country = 'Italy'
  AND lt.milliseconds = (
      SELECT MIN(lt2.milliseconds)
      FROM lapTimes lt2
      JOIN races r2 ON lt2.raceId = r2.raceId
      WHERE r2.circuitId = c.circuitId
  )
ORDER BY c.name, lt.milliseconds;

```

```sql
WITH italian_laps AS (
    SELECT 
        c.circuitId,
        c.name AS circuit_name,
        c.location,
        lt.time,
        lt.milliseconds,
        lt.driverId,
        r.year
    FROM circuits c
    JOIN races r ON c.circuitId = r.circuitId
    JOIN lapTimes lt ON r.raceId = lt.raceId
    WHERE c.country = 'Italy'
),
fastest AS (
    SELECT circuitId, MIN(milliseconds) AS min_ms
    FROM italian_laps
    GROUP BY circuitId
)
SELECT 
    il.circuit_name,
    il.location,
    il.time AS lap_record,
    il.milliseconds,
    d.forename || ' ' || d.surname AS driver,
    il.year AS race_year
FROM italian_laps il
JOIN fastest f ON il.circuitId = f.circuitId AND il.milliseconds = f.min_ms
JOIN drivers d ON il.driverId = d.driverId
ORDER BY il.circuit_name;

```
