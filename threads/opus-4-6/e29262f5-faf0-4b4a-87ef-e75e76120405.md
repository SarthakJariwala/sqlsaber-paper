SELECT 
    t.account_id,
    t.client_id,
    t.birth_date,
    t.average_salary AS lowest_avg_salary,
    (SELECT MAX(A11) FROM district) AS highest_avg_salary,
    (SELECT MAX(A11) FROM district) - t.average_salary AS salary_gap
FROM (
    SELECT c.client_id, c.birth_date, d.A11 AS average_salary, disp.account_id
    FROM client c
    JOIN district d ON c.district_id = d.district_id
    JOIN disp ON c.client_id = disp.client_id

) t
ORDER BY t.average_salary ASC
