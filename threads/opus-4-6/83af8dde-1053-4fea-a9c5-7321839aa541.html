SELECT 
    CAST(COUNT(T1.id) AS REAL) / 4 AS annual_avg_sets,
    (
        SELECT T3.language 
        FROM cards T2 
        INNER JOIN foreign_data T3 ON T2.uuid = T3.uuid
        GROUP BY T3.language 
        ORDER BY COUNT(T3.language) DESC 
        LIMIT 1
    ) AS most_common_language
FROM sets T1
WHERE T1.releaseDate BETWEEN '2012-01-01' AND '2015-12-31'
