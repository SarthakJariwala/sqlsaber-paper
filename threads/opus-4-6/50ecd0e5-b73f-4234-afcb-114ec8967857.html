SELECT 
    a.element, 
    COUNT(a.element) AS tally
FROM atom a
JOIN molecule m ON a.molecule_id = m.molecule_id
WHERE m.label = '+'
  AND SUBSTR(a.atom_id, 7, 1) = '4'
GROUP BY a.element
ORDER BY tally DESC;
