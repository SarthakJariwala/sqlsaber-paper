```sql
SELECT 
    ra.year,
    AVG(r.milliseconds / 1000.0) AS avg_time_seconds
FROM results r
JOIN races ra ON r.raceId = ra.raceId
WHERE r.position = 1 
  AND r.time IS NOT NULL 
  AND ra.year < 1975
GROUP BY ra.year
ORDER BY ra.year

```
