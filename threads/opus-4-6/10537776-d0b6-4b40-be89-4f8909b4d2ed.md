```sql

SELECT 
    s.School,
    sat.NumGE1500,
    s.AdmFName1 || ' ' || s.AdmLName1 AS Administrator1,
    s.AdmFName2 || ' ' || s.AdmLName2 AS Administrator2,
    s.AdmFName3 || ' ' || s.AdmLName3 AS Administrator3
FROM satscores sat
JOIN schools s ON sat.cds = s.CDSCode
WHERE sat.NumGE1500 IS NOT NULL
ORDER BY sat.NumGE1500 DESC
LIMIT 1;

```
