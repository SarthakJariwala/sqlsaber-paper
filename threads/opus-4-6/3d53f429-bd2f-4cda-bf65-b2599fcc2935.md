-- Step 1: Get total annual consumption per customer for 2013, CZK only
-- Step 2: Find the minimum consumption per segment
-- Step 3: Count customers with that min consumption and compute annual avg
-- Step 4: Compute differences

WITH customer_annual AS (
    SELECT 
        c.Segment,
        ym.CustomerID,
        SUM(ym.Consumption) AS total_consumption
    FROM yearmonth ym
    JOIN customers c ON ym.CustomerID = c.CustomerID
    WHERE c.Currency = 'CZK'
      AND ym.Date BETWEEN '201301' AND '201312'
    GROUP BY c.Segment, ym.CustomerID
),
min_per_segment AS (
    SELECT 
        Segment,
        MIN(total_consumption) AS min_consumption
    FROM customer_annual
    GROUP BY Segment
),
lowest_customers AS (
    SELECT 
        ca.Segment,
        SUM(ca.total_consumption) AS total_of_lowest,
        COUNT(*) AS cnt_lowest
    FROM customer_annual ca
    JOIN min_per_segment ms ON ca.Segment = ms.Segment AND ca.total_consumption = ms.min_consumption
    GROUP BY ca.Segment
),
annual_avg AS (
    SELECT 
        Segment,
        total_of_lowest / cnt_lowest AS annual_average
    FROM lowest_customers
)
SELECT 
    sme.annual_average AS SME_avg,
    lam.annual_average AS LAM_avg,
    kam.annual_average AS KAM_avg,
    ROUND(sme.annual_average - lam.annual_average, 2) AS "SME - LAM",
    ROUND(lam.annual_average - kam.annual_average, 2) AS "LAM - KAM",
    ROUND(kam.annual_average - sme.annual_average, 2) AS "KAM - SME"
FROM 
    (SELECT annual_average FROM annual_avg WHERE Segment = 'SME') sme,
    (SELECT annual_average FROM annual_avg WHERE Segment = 'LAM') lam,
    (SELECT annual_average FROM annual_avg WHERE Segment = 'KAM') kam;
