FROM superhero s

JOIN publisher p ON s.publisher_id = p.id

LEFT JOIN colour hc ON s.hair_colour_id = hc.id
LEFT JOIN colour sc ON s.skin_colour_id = sc.id
WHERE g.gender = 'Female'
  AND p.publisher_name = 'Dark Horse Comics';
