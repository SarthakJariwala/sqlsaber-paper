```sql

SELECT COUNT(DISTINCT P.ID) AS num_male_patients
FROM Patient P
INNER JOIN Laboratory L ON P.ID = L.ID
WHERE strftime('%Y', P.Birthday) BETWEEN '1936' AND '1956'
  AND P.SEX = 'M'
```
