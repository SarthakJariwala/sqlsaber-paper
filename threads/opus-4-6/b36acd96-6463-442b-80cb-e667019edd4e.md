    P.ID,
    L.UA,

    E.`aCL IgA`
FROM Patient P
INNER JOIN Laboratory L ON P.ID = L.ID
INNER JOIN Examination E ON P.ID = E.ID
WHERE P.SEX = 'F'
  AND L.UA <= 6.50
ORDER BY L.UA DESC
LIMIT 1;
